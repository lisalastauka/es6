ES6 & ES2017

Overview and features

Created by Elizabeth Dolgaya

ES6

  • It became a standard on 17 June 2015.
  • Most of its features are already widely available (as documented in kangax’ ES6 compatibility table).
  • Transpilers (such as Babel) let you compile ES6 to ES5.

Goals for ES6:

Be a better language

for writing:

  • complex applications;
  • libraries (possibly including the DOM) shared by those applications;
  • code generators targeting the new edition.

Improve interoperation

adopting de facto standards where possible:

  • Classes
  • Modules
  • Arrow functions
  • Named function parameters

Versioning

as simple and linear as possible.

ES6 features

Better syntax for existing features:

  • Classes
  • Modules

New functionality in the standard library:

  • New methods for strings and Arrays
  • Promises
  • Maps, Sets

New features:

  • Generators
  • Proxies
  • WeakMaps

From var to const/let


//In ES5				
var x = 3;
function func(randomize) {
    if (randomize) {
        var x = Math.random(); // (A) scope: whole function
        return x;
    }
    return x; // accesses the x from line A
}
func(false); // undefined
					

//In ES6
let x = 3;
function func(randomize) {
    if (randomize) {
        let x = Math.random();
        return x;
    }
    return x;
}
func(false); // 3

Tips:

  • Const – for variables whose values never change.
  • Let – for variables whose values do change.
  • Avoid var.

From IIFEs to blocks


//In ES5				
(function () {  // open IIFE
    var tmp = ···;
    ···
}());  // close IIFE

console.log(tmp); // ReferenceError
					

//In ES6
{  // open block
    let tmp = ···;
    ···
}  // close block

console.log(tmp); // ReferenceError

From concatenating to template literals



//In ES5				
function printCoord(x, y) {
    console.log('('+x+', '+y+')');
}
					

//In ES6					
function printCoord(x, y) {
    console.log(`(${x}, ${y})`);
}

From function expressions to arrow functions



//In ES5				
var arr = [1, 2, 3];
var squares = arr.map(function (x) { return x * x });

					

//In ES6					
const arr = [1, 2, 3];
const squares = arr.map(x => x * x);

From for to forEach() to for-of


//Prior to ES5				
var arr = ['a', 'b', 'c'];
for (var i=0; i < arr.length; i++) {
    var elem = arr[i];
    console.log(elem);
}

					

//In ES5				
arr.forEach(function (elem) {
    console.log(elem);
});

					

//In ES6					
const arr = ['a', 'b', 'c'];
for (const elem of arr) {
    console.log(elem);
}

From arguments to rest parameters



//In ES5

    for (var i=0; i < arguments.length; i++) {
        console.log(arguments[i]);
    }
}
					

//In ES6

        console.log(arg);
    }
}

//In ES6

Array.prototype.push()


//In ES5				
var arr1 = ['a', 'b'];
var arr2 = ['c', 'd'];

arr1.push.apply(arr1, arr2);
    // arr1 is now ['a', 'b', 'c', 'd']
					

//In ES6					
const arr1 = ['a', 'b'];
const arr2 = ['c', 'd'];

arr1.push(...arr2);
    // arr1 is now ['a', 'b', 'c', 'd']

From constructors to classes

Base classes


//In ES5				
function Person(name) {
    this.name = name;
}
Person.prototype.describe = function () {
    return 'Person called '+this.name;
};
					

//In ES6					
class Person {
    constructor(name) {
        this.name = name;
    }
    describe() {
        return 'Person called '+this.name;
    }
}

Derived classes


//In ES5				
function Employee(name, title) {
    Person.call(this, name); // super(name)
    this.title = title;
}
Employee.prototype = Object.create(Person.prototype);
Employee.prototype.constructor = Employee;
Employee.prototype.describe = function () {
    return Person.prototype.describe.call(this) // super.describe()
           + ' (' + this.title + ')';
};
					

//In ES6					
class Employee extends Person {
    constructor(name, title) {
        super(name);
        this.title = title;
    }
    describe() {
        return super.describe() + ' (' + this.title + ')';
    }
}

From custom error constructors to subclasses of Error


//In ES5				
function MyError() {
    // Use Error as a function
    var superInstance = Error.apply(null, arguments);
    copyOwnPropertiesFrom(this, superInstance);
}
MyError.prototype = Object.create(Error.prototype);
MyError.prototype.constructor = MyError;

function copyOwnPropertiesFrom(target, source) {
    Object.getOwnPropertyNames(source)
    .forEach(function(propKey) {
        var desc = Object.getOwnPropertyDescriptor(source, propKey);
        Object.defineProperty(target, propKey, desc);
    });
    return target;
};
					

//In ES6					
class MyError extends Error {
}

From objects to Maps


//In ES5
var dict = Object.create(null);
function countWords(word) {
    var escapedWord = escapeKey(word);
    if (escapedWord in dict) {
        dict[escapedWord]++; } else {
        dict[escapedWord] = 1;
    }
}
function escapeKey(key) {
    if (key.indexOf('__proto__') === 0) {
        return key+'%'; } else {
        return key;
    }
}
					

//In ES6					
const map = new Map();
function countWords(word) {
    const count = map.get(word) || 0;
    map.set(word, count + 1);
}

New string methods

From indexOf to startsWith:


if (str.indexOf('x') === 0) {} // ES5

if (str.startsWith('x')) {} // ES6

From indexOf to endsWith:


function endsWith(str, suffix) { // ES5
  var index = str.indexOf(suffix);
  return index >= 0
    && index === str.length-suffix.length;
}

str.endsWith(suffix); // ES6

From indexOf to includes:


if (str.indexOf('x') >= 0) {} // ES5

if (str.includes('x')) {} // ES6

From join to repeat:


new Array(3+1).join('#') // ES5

'#'.repeat(3) // ES6

New Array methods

From Array.prototype.indexOf to Array.prototype.findIndex:


const arr = ['a', NaN];

arr.indexOf(NaN); // -1 
arr.findIndex(x => Number.isNaN(x)); // 1


> isNaN('abc')
true
> Number.isNaN('abc')
false

From Array.prototype.slice() to Array.from() or the spread operator:


var arr1 = Array.prototype.slice.call(arguments); // ES5
const arr2 = Array.from(arguments); // ES6


const arr1 = [...'abc'];
    // ['a', 'b', 'c']
const arr2 = [...new Set().add('a').add('b')];
    // ['a', 'b']

From apply() to Array.prototype.fill():


// Same as Array(undefined, undefined)
var arr1 = Array.apply(null, new Array(2));
    // [undefined, undefined]

const arr2 = new Array(2).fill(undefined);
    // [undefined, undefined]
					

// ES5
var arr3 = Array.apply(null, new Array(2))
    .map(function (x) { return 'x' });
    // ['x', 'x']

// ES6
const arr4 = new Array(2).fill('x');
    // ['x', 'x']

Multiple exports

Multiple exports in CommonJS


//------ lib.js ------
var sqrt = Math.sqrt;
function square(x) {
    return x * x;
}

module.exports = {
    sqrt: sqrt,
    square: square,
};

//------ main1.js ------
var square = require('lib').square;

console.log(square(11)); // 121

					

//------ main2.js ------
var lib = require('lib');
console.log(lib.square(11)); // 121

Multiple exports in ES6


//------ lib.js ------
export const sqrt = Math.sqrt;
export function square(x) {
    return x * x;
}
export function diag(x, y) {
    return sqrt(square(x) + square(y));
}

//------ main1.js ------
import { square, diag } from 'lib';
console.log(square(11)); // 121
console.log(diag(4, 3)); // 5

					

//------ main2.js ------
import * as lib from 'lib'; // (A)
console.log(lib.square(11)); // 121
console.log(lib.diag(4, 3)); // 5
